// Pour la commande draw (Point -- Point){1,}
// Point représente des coordonnées polaires ou cartésiennes en dur (donc non calculées par des expressions)
// Les espaces, c'est juste pour que ça soit lisible :)

/* Arbre de dérivation : ok */

S --> M $
M --> draw A
M -> epsi
A --> ( E ) N
E --> digit E' digit
E' --> :
E' --> ,
N --> sep A
N --> epsi


/* => seul ce qui change est reporté */

// Pour la commande draw (Point -- Point)
// Point représente des coordonnées polaires ou cartésiennes calculées via des expressions

/* Arbre de dérivation : ok */

A --> ( P ) N
P --> E E' E
E --> digit E"
E --> ( digit E" )
E" --> O E
E" --> epsi
O --> + | * | - | /


// Pour la commande draw (Point -- +Point){1,} => les coordonnées relatives

/* Arbre de dérivation : ok */

N --> sep R
R --> + A
R --> A


// Pour la commande draw (Point -- Point -- cycle) => pour gérer les cycles

R --> cycle N

// Pour les commandes fill, draw, image

/* Arbre de dérivation : ok */

M --> I
M --> image { I }
I --> draw A
I --> fill A
I --> epsi

//Grammaire globale => permet de reconnaitre toutes les commandes à implémenter

/* A TESTER DANS LA PRATIQUE */

// Ne gère pas les multiples instructions !

S --> M $
M --> I | image { I };
I --> draw A; | fill A; | epsi
A --> ( P ) N
P --> E E' E
E --> digit E" | ( digit E" )
E" --> O E | epsi
O --> * | + | * | /
N --> sep R
R --> +A | A | cycle N
