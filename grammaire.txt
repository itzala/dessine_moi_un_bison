// Pour la commande draw (Point -- Point){1,}
// Point représente des coordonnées polaires ou cartésiennes en dur (donc non calculées par des expressions)
// Les espaces, c'est juste pour que ça soit lisible :)

S --> M $
M --> draw A
M -> epsi
A --> ( E ) N
E --> digit E' digit
E' --> :
E' --> ,
N --> sep A
N --> epsi

// Pour la commande draw (Point -- Point)
// Point représente des coordonnées polaires ou cartésiennes calculées via des expressions

E --> X E' X
X --> digit
X --> (X O X)
O --> +
O --> /
O --> *
O --> -


// Pour la commande draw (Point -- +Point){1,} => les coordonnées relatives

A --> T ( E ) N
T --> +
T --> epsi


// Pour la commande draw (Point -- Point -- cycle) => pour gérer les cycles

N --> sep C A
C --> cycle N
C --> epsi
